function TypingDots() {
  return (
    <div style={{ display: "flex", alignItems: "center", gap: 6, padding: "12px 16px" }}>
      {[0, 1, 2].map((i) => (
        <div key={i} style={{ width: 8, height: 8, borderRadius: "50%", background: "#94a3b8", animation: `dot 1.2s ease-in-out ${i * 0.15}s infinite` }} />
      ))}
    </div>
  );
}

function Bubble({ role, content }) {
  const isUser = role === "user";
  return (
    <div style={{ display: "flex", justifyContent: isUser ? "flex-end" : "flex-start", marginBottom: 10, animation: "fadeUp 0.25s ease" }}>
      {!isUser && (
        <div style={{ width: 34, height: 34, borderRadius: "50%", background: "linear-gradient(135deg,#2d6a4f,#40916c)", display: "flex", alignItems: "center", justifyContent: "center", fontSize: 17, flexShrink: 0, marginRight: 8, marginTop: 4, boxShadow: "0 2px 8px rgba(45,106,79,0.25)" }}>🌍</div>
      )}
      <div style={{ maxWidth: "80%", padding: "10px 14px", borderRadius: isUser ? "16px 16px 4px 16px" : "16px 16px 16px 4px", background: isUser ? "linear-gradient(135deg,#2d6a4f,#40916c)" : "var(--bb)", color: isUser ? "#fff" : "var(--tx)", fontSize: 14.5, lineHeight: 1.7, wordBreak: "break-word", whiteSpace: "pre-wrap", boxShadow: isUser ? "0 2px 10px rgba(45,106,79,0.2)" : "0 1px 4px rgba(0,0,0,0.05)" }}>{content}</div>
    </div>
  );
}

export default function App() {
  const [lang, setLang] = useState(null);
  const [terrain, setTerrain] = useState(null);
  const [messages, setMessages] = useState([]);
  const [input, setInput] = useState("");
  const [loading, setLoading] = useState(false);
  const [started, setStarted] = useState(false);
  const endRef = useRef(null);

  useEffect(() => { endRef.current?.scrollIntoView({ behavior: "smooth" }); }, [messages, loading]);

  const send = async (text) => {
    if (!text.trim() || loading) return;
    setStarted(true);
    const userMsg = { role: "user", content: text.trim() };
    const next = [...messages, userMsg];
    setMessages(next);
    setInput("");
    setLoading(true);

    // Gemini API 연동 설정
    const apiKey = ""; // 런타임에서 자동으로 주입됩니다.
    const url = `https://generativelanguage.googleapis.com/v1beta/models/gemini-2.5-flash-preview-09-2025:generateContent?key=${apiKey}`;
    
    // Gemini 메시지 포맷으로 변환
    const geminiContents = next.map(m => ({
      role: m.role === "assistant" ? "model" : "user",
      parts: [{ text: m.content }]
    }));

    const payload = {
      contents: geminiContents,
      systemInstruction: {
        parts: [{ text: getSystemPrompt(lang, terrain) }]
      }
    };

    // 지수 백오프 기반 재시도 로직
    const delays = [1000, 2000, 4000, 8000, 16000];
    let attempt = 0;
    let successText = "";

    while (attempt < delays.length) {
      try {
        const res = await fetch(url, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify(payload),
        });

        if (!res.ok) throw new Error("API call failed");

        const data = await res.json();
        successText = data.candidates?.[0]?.content?.parts?.[0]?.text;
        
        if (successText) {
          setMessages((p) => [...p, { role: "assistant", content: successText }]);
          break; // 성공 시 루프 종료
        } else {
          throw new Error("No content received");
        }
      } catch (error) {
        if (attempt === delays.length - 1) {
          // 모든 재시도 실패 시
          setMessages((p) => [...p, { role: "assistant", content: "인터넷을 확인하거나 다시 질문해 줘! 🔌" }]);
        } else {
          // 실패 시 대기 후 재시도
          await new Promise(r => setTimeout(r, delays[attempt]));
          attempt++;
        }
      }
    }
    
    setLoading(false);
  };

  const reset = () => { setLang(null); setTerrain(null); setMessages([]); setStarted(false); };

  const dk = typeof window !== "undefined" && window.matchMedia?.("(prefers-color-scheme:dark)").matches;
  const vars = { "--bg": dk ? "#0f1a14" : "#f7faf8", "--bb": dk ? "#1e3a2a" : "#f0fdf4", "--tx": dk ? "#d4e7dc" : "#1e293b", "--tx2": dk ? "#8aac9a" : "#64748b", "--bd": dk ? "#2a4a3a" : "#d1e7dd", "--inp": dk ? "#162920" : "#fff" };
  const css = `@import url('https://cdn.jsdelivr.net/gh/orioncactus/pretendard/dist/web/static/pretendard.css');@keyframes fadeUp{from{opacity:0;transform:translateY(8px)}to{opacity:1;transform:translateY(0)}}@keyframes float{0%,100%{transform:translateY(0)}50%{transform:translateY(-6px)}}@keyframes dot{0%,60%,100%{transform:translateY(0);opacity:.4}30%{transform:translateY(-8px);opacity:1}}*{box-sizing:border-box;margin:0;padding:0}`;

  // --- Language ---
  if (!lang) {
    return (
      <div style={{ ...vars, height: "100vh", display: "flex", flexDirection: "column", alignItems: "center", justifyContent: "center", background: "var(--bg)", fontFamily: "'Pretendard','Noto Sans KR',sans-serif", gap: 32 }}>
        <style>{css}</style>
        <div style={{ textAlign: "center", animation: "fadeUp 0.5s ease" }}>
          <div style={{ fontSize: 56, marginBottom: 16, animation: "float 3s ease-in-out infinite" }}>🗺️</div>
          <div style={{ fontSize: 22, fontWeight: 700, color: "var(--tx)", marginBottom: 8 }}>지형 탐험 도우미</div>
          <div style={{ fontSize: 14, color: "var(--tx2)", lineHeight: 1.6 }}>언어를 선택하세요<br />选择语言 · Выберите язык</div>
        </div>
        <div style={{ display: "flex", gap: 12, animation: "fadeUp 0.6s ease" }}>
          {LANGUAGES.map((l) => (
            <button key={l.code} onClick={() => setLang(l.code)} style={{ display: "flex", flexDirection: "column", alignItems: "center", gap: 8, padding: "20px 28px", border: "2px solid var(--bd)", borderRadius: 16, background: "var(--inp)", cursor: "pointer", transition: "all 0.2s" }}
              onMouseEnter={(e) => { e.currentTarget.style.borderColor = "#2d6a4f"; e.currentTarget.style.transform = "translateY(-3px)"; e.currentTarget.style.boxShadow = "0 6px 20px rgba(45,106,79,0.15)"; }}
              onMouseLeave={(e) => { e.currentTarget.style.borderColor = "var(--bd)"; e.currentTarget.style.transform = "translateY(0)"; e.currentTarget.style.boxShadow = "none"; }}
            >
              <span style={{ fontSize: 32 }}>{l.flag}</span>
              <span style={{ fontSize: 15, fontWeight: 600, color: "var(--tx)" }}>{l.label}</span>
            </button>
          ))}
        </div>
      </div>
    );
  }

  // --- Terrain ---
  if (!terrain) {
    const terrainList = TERRAINS[lang] || TERRAINS.ko;
    const pickLabel = (LABELS[lang] || LABELS.ko).pick;
    return (
      <div style={{ ...vars, height: "100vh", display: "flex", flexDirection: "column", alignItems: "center", justifyContent: "center", background: "var(--bg)", fontFamily: "'Pretendard','Noto Sans KR',sans-serif", gap: 32 }}>
        <style>{css}</style>
        <div style={{ textAlign: "center", animation: "fadeUp 0.5s ease" }}>
          <div style={{ fontSize: 56, marginBottom: 16, animation: "float 3s ease-in-out infinite" }}>🗺️</div>
          <div style={{ fontSize: 18, fontWeight: 700, color: "var(--tx)", marginBottom: 6 }}>{pickLabel}</div>
        </div>
        <div style={{ display: "flex", gap: 12, animation: "fadeUp 0.6s ease" }}>
          {terrainList.map((t) => (
            <button key={t.code} onClick={() => setTerrain(t.code)} style={{ display: "flex", flexDirection: "column", alignItems: "center", gap: 8, padding: "20px 28px", border: "2px solid var(--bd)", borderRadius: 16, background: "var(--inp)", cursor: "pointer", transition: "all 0.2s" }}
              onMouseEnter={(e) => { e.currentTarget.style.borderColor = "#2d6a4f"; e.currentTarget.style.transform = "translateY(-3px)"; e.currentTarget.style.boxShadow = "0 6px 20px rgba(45,106,79,0.15)"; }}
              onMouseLeave={(e) => { e.currentTarget.style.borderColor = "var(--bd)"; e.currentTarget.style.transform = "translateY(0)"; e.currentTarget.style.boxShadow = "none"; }}
            >
              <span style={{ fontSize: 36 }}>{t.emoji}</span>
              <span style={{ fontSize: 16, fontWeight: 600, color: "var(--tx)" }}>{t.label}</span>
            </button>
          ))}
        </div>
        <button onClick={() => setLang(null)} style={{ fontSize: 13, color: "var(--tx2)", background: "none", border: "none", cursor: "pointer", textDecoration: "underline" }}>
          ← {lang === "ko" ? "언어 다시 선택" : lang === "zh" ? "重新选择语言" : "Сменить язык"}
        </button>
      </div>
    );
  }

  // --- Chat ---
  const terrainObj = (TERRAINS[lang] || TERRAINS.ko).find((t) => t.code === terrain);
  const langObj = LANGUAGES.find((l) => l.code === lang);
  const entryQ = ENTRY_Q[terrain]?.[lang] || ENTRY_Q[terrain]?.ko;
  const placeholder = lang === "ko" ? "궁금한 것을 물어봐!" : lang === "zh" ? "请输入问题..." : "Задайте вопрос...";
  const startLabel = (LABELS[lang] || LABELS.ko).start;

  return (
    <div style={{ ...vars, height: "100vh", display: "flex", flexDirection: "column", background: "var(--bg)", fontFamily: "'Pretendard','Noto Sans KR',sans-serif", overflow: "hidden" }}>
      <style>{css}</style>

      {/* Header */}
      <div style={{ background: "linear-gradient(135deg,#2d6a4f,#1b4332)", padding: "14px 16px", color: "#fff", flexShrink: 0, display: "flex", alignItems: "center", justifyContent: "space-between" }}>
        <div style={{ display: "flex", alignItems: "center", gap: 10 }}>
          <span style={{ fontSize: 24 }}>{terrainObj?.emoji}</span>
          <div>
            <div style={{ fontSize: 16, fontWeight: 700 }}>{terrainObj?.label} {lang === "ko" ? "탐험" : lang === "zh" ? "探索" : "исследование"}</div>
            <div style={{ fontSize: 11, opacity: 0.8, marginTop: 1 }}>{langObj?.flag} {langObj?.label}</div>
          </div>
        </div>
        <button onClick={reset} style={{ padding: "6px 12px", borderRadius: 8, border: "1px solid rgba(255,255,255,0.3)", background: "rgba(255,255,255,0.1)", color: "#fff", fontSize: 12, cursor: "pointer" }}>
          {lang === "ko" ? "처음으로" : lang === "zh" ? "重新开始" : "Сначала"}
        </button>
      </div>

      {/* Chat area */}
      <div style={{ flex: 1, overflowY: "auto", padding: "12px 14px 6px" }}>
        {/* Entry button */}
        {!started && (
          <div style={{ animation: "fadeUp 0.4s ease", textAlign: "center", paddingTop: 32 }}>
            <div style={{ fontSize: 48, marginBottom: 16, animation: "float 3s ease-in-out infinite" }}>{terrainObj?.emoji}</div>
            <button
              onClick={() => send(entryQ)}
              style={{
                display: "inline-flex", alignItems: "center", gap: 10,
                padding: "14px 24px", border: "2px solid #2d6a4f", borderRadius: 16,
                background: "linear-gradient(135deg,#2d6a4f,#40916c)", color: "#fff",
                fontSize: 16, fontWeight: 600, cursor: "pointer", transition: "all 0.2s",
                boxShadow: "0 4px 16px rgba(45,106,79,0.3)",
              }}
              onMouseEnter={(e) => { e.currentTarget.style.transform = "translateY(-2px)"; e.currentTarget.style.boxShadow = "0 6px 24px rgba(45,106,79,0.4)"; }}
              onMouseLeave={(e) => { e.currentTarget.style.transform = "translateY(0)"; e.currentTarget.style.boxShadow = "0 4px 16px rgba(45,106,79,0.3)"; }}
            >
              <span>{entryQ}</span>
            </button>
            <div style={{ fontSize: 13, color: "var(--tx2)", marginTop: 12 }}>{startLabel}</div>
          </div>
        )}

        {messages.map((m, i) => <Bubble key={i} role={m.role} content={m.content} />)}
        {loading && <TypingDots />}
        <div ref={endRef} />
      </div>

      {/* Input */}
      <div style={{ padding: "10px 14px 14px", borderTop: "1px solid var(--bd)", background: "var(--bg)", flexShrink: 0 }}>
        <div style={{ display: "flex", gap: 8 }}>
          <input value={input} onChange={(e) => setInput(e.target.value)}
            onKeyDown={(e) => { if (e.key === "Enter" && !e.shiftKey) { e.preventDefault(); send(input); } }}
            placeholder={placeholder} disabled={loading}
            style={{ flex: 1, padding: "11px 14px", border: "1.5px solid var(--bd)", borderRadius: 12, fontSize: 14.5, background: "var(--inp)", color: "var(--tx)", outline: "none", transition: "border-color 0.2s" }}
            onFocus={(e) => (e.target.style.borderColor = "#2d6a4f")}
            onBlur={(e) => (e.target.style.borderColor = "var(--bd)")}
          />
          <button onClick={() => send(input)} disabled={loading || !input.trim()}
            style={{ width: 42, height: 42, borderRadius: 12, border: "none", background: loading || !input.trim() ? "#94a3b8" : "linear-gradient(135deg,#2d6a4f,#40916c)", color: "#fff", fontSize: 18, cursor: loading || !input.trim() ? "default" : "pointer", display: "flex", alignItems: "center", justifyContent: "center", flexShrink: 0, transition: "all 0.2s" }}
          >↑</button>
        </div>
      </div>
    </div>
  );
}
